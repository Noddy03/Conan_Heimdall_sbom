cmake_minimum_required(VERSION 3.15)
project(HeimdallAdapters)

# Set C++ standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Add subdirectories for your adapter libraries
add_subdirectory(src/lld)
add_subdirectory(src/gold)

# Add the executable
add_executable(heimdall_app src/main.cpp)

# Link the adapter libraries to the executable
target_link_libraries(heimdall_app PRIVATE lld_adapter gold_adapter)
