cmake_minimum_required(VERSION 3.20)

# Project name and C++ standard
project(HeimdallAdapters VERSION 1.0 LANGUAGES CXX)
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Include subdirectories for libraries
add_subdirectory(src/lld)
add_subdirectory(src/golden)

# Add main executable
add_executable(main_exe src/main.cpp)

# Link the adapter libraries to the executable
target_link_libraries(main_exe PRIVATE LldAdapter GoldAdapter)
